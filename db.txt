CREATE TABLE `scrna_sys`. (`id` INT(100) NOT NULL AUTO_INCREMENT , `name` VARCHAR(255) NULL DEFAULT NULL , `filename` VARCHAR(255) NULL DEFAULT NULL, `meta_filename` VARCHAR(255) NULL DEFAULT NULL , `tissue` VARCHAR(255) NULL DEFAULT NULL , `publish_year` YEAR(4) NOT NULL DEFAULT CURRENT_TIMESTAMP , `status` ENUM('available', 'unavailable') NULL DEFAULT 'unavailable' , UNIQUE `id` (`id`)) ENGINE = InnoDB;
